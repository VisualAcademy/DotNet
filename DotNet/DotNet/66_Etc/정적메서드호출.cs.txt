using System;

namespace 정적메서드호출{
	public class 안녕클래스{
		public static void 안녕하세요(){
			Console.WriteLine("안녕하세요!!!");
		}
	}  
	public class 메롱클래스{
		public static void 메롱(){
			Console.WriteLine("메롱~~~");
		}
	}

	public class 정적메서드호출{
		public static void Main(){
			//정적(static) 메서드 호출
			안녕클래스.안녕하세요();
			메롱클래스.메롱();
		}
	}
}
