// 수열합계 : 1 - 2 + 4 - 7 + 11 - 16 + 22 = 13 
// 위 패턴대로 7번만 반복 -> 100번 반복한다면???
using System;

public class 합계알고리즘적용
{
  public static void Main()
  {
    int sum = 0; // 13
    int data = 1; // 1부터 시작
    for (int i = 0; i < 7; i++)
    {
      if (i % 2 == 0) // 짝수라면 
      {
        sum += data;//누적  
      }
      else
      {
        sum -= data;//가감
      }
      data += (i + 1); // 다음번 인덱스값을 누적
    }
    Console.WriteLine(sum);//13
  }
}
