// ASP.NET 2.0 전용
// 무명 메서드 : 대리자에 의해서 호출될 메서드를 따로 만들지 않고, 
// 대리자를 인스턴스화 할 때 메서드 코드 블록을 직접 지정하는 방법
using System;

class 무명메서드
{
	//[1] 델리게이트 선언
	public delegate void Write(string s);

	static void Main(string[] args)
	{
		//[2] 무명 메서드 선언
		Write write = delegate(string s) { Console.Write(s); };

		//[3] 무명 메서드 호출
		write("안녕하세요.\t");
		write("반갑습니다.\n");
	}
}
