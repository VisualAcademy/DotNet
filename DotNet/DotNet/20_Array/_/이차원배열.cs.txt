using System;

public class 이차원배열
{
  public static void Main()
  { 

    //[1] 이차원 배열 선언 : int a[3][2]
    int[,] a = new int[3, 2] 
      {
        {1,2},
        {3,4},
        {5,6}
      };
    // 행 반복
    for (int i = 0; i < 3; i++)
    {
      // 열 반복
      for (int j = 0; j < 2; j++)
      {
        Console.Write("{0}\t", a[i,j]);  
      }
      Console.WriteLine();
    }

  }
}
